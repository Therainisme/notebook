# useImperativeHandle

一般与forwardRef配合使用，通过ref将子组件的方法暴露给父组件

子组件

function Children(props : IProp, ref: React.Ref<unknown> | undefined) {
    useImperativeHandle(ref, () => ({
        displayMsg
    }), [])

    function displayMsg(msg: string) {
        // 被暴露的函数
    }
}

// 别忘记包裹forwardRef
export default forwardRef(Children);
